013    /**
014     * A <b>Predicate</b> is a named list of <b>Argument</b>s.  An argument can be
015     * either an <i>Identifier</i> or a <i>Constant</i>.
016     *
017     * <p>
018     * Syntax: <b>Predicate</b> &rarr;
019     *              <i>Identifier</i> "(" <b>Argument</b> [ "," <b>Argument</b> ]*
020     *
021     * <p>
022     * There are four properties that we need to define for a <b>Predicate</b>
023     *
024     * <ol>
025     *    <li>
026     *    A <b>Predicate</b> contains a list of arguments.  It is implemented
027     *    as an <b>ArrayList<Node></b>.  We enforce the fact that is is a list of
028     *    arguments by only creating elements for the list using the
029     *    <i>createNode(Lex)</i> method.  The <i>createNode(Lex)</i> method below
030     *    creates only arguments. An <b>Argument</b> is an instance of type
031     *    <b>Node</b> so it can be added to an ArrayList<Node>.
032     *    </li>
033     *
034     *    <li>
035     *    To determine whether to parse another <b>Argument</b> and add it to the
036     *    list we need to know whether lex.front() is in the first of
037     *    <b>Argument</b>.  This is implemented by the method
038     *    <i>isInFirstOfListElement(Lex)</i> below.
039     *    </li>
040     * 
041     *    <li>
042     *    A <b>Predicate</b> is a list of one-or-more elements.  This is the
043     *    default value of a list and defined in <i>List::getLengthConstraint()</i>.
044     *    We do NOT override it here.
045     *    </li>
046     * 
047     *    <li>
048     *    The elements of a <b>Predicate</b>'s constant list are separated by 
049     *    commas. This is defined in <i>NamedList::getSeparator()</i>.  We do NOT
050     *    override it here.
051     *    </li>
052     * </ol>
053     *
054     * <pre>
055     * Domain Definition:
056     *    predicateName : Identifier 
057     *    argumentList  : Sequence&lt;<b>Argument</b>&gt;
058     *
059     * Instance Invariant:
060     *     predicateName &ne; null  AND
061     *     |predicateName| &gt; 0 AND
062     *     |argumentList| &gt; 0
063     *
064     * </pre>
065     *
066     * Last Modified: 21 Feb 2006
067     *
068     * @author Scott N. Woodfield
069     * @version 1.0
070     * @see
071     *   <a href=
072     *      "../../srchtml/project3/Predicate.html"
073     *      >Source Code
074     *   </a>
075     */

178        public Predicate(Predicate predicate)
179            throws ParserException
180        {
181            super();
182            Iterator<Node> iter = predicate.nodes.iterator();
183            while(iter.hasNext()){
184                nodes.add(Argument.createCopy((Argument)iter.next()));
185            };
186            name = new Identifier(predicate.name);
187        }

190        /**
191         * Checks to see if all of the arguments are <i>Identifier</i>s.
192         *
193         * <p>
194         * <table>
195         *   <tr>
196         *     <td valign=top>
197         *       <b>Precondition</b>:
198         *     </td>
199         *     <td valign=top>
200         *       none
201         *     </td>
202         *     <td valign=top>
203         *     </td>
204         *   </tr>
205         *   <tr>
206         *     <td valign=top>
207         *       <b>Postcondition</b>:
208         *     </td>
209         *     <td valign=top>
210         *       result = &forall;<sub>x</sub>  (x &isin; nodes &rArr; 
211         *                x <i>instanceof</i> <i>Constant</i>)
212         *     </td>
213         *     <td valign=top>
214         *     </td>
215         *   </tr>
216         * </table>
217         *
218         * @return whether all of the nodes are <i>Constant</i>s.
219         * @see
220         *   <a href=
221         *      "../../srchtml/project3/Predicate.html#allConstants"
222         *      >Source Code
223         *   </a>

225        public boolean allConstants()
226        {
227            boolean result = true;
228            Iterator<Node> iter = nodes.iterator();
229            while(result && iter.hasNext()){
230                result = iter.next() instanceof Constant;
231            };
232            return result;
233        }

235        /**
236         * Returns a hash code for this predicate.
237         *
238         * <p>
239         * <table>
240         *   <tr>
241         *     <td valign=top>
242         *       <b>Precondition</b>:
243         *     </td>
244         *     <td valign=top>
245         *       none
246         *     </td>
247         *     <td valign=top>
248         *     </td>
249         *   </tr>
250         *   <tr>
251         *     <td valign=top>
252         *       <b>Postcondition</b>:
253         *     </td>
254         *     <td valign=top>
255         *        result = (int) &sum; 0 &lt;= i &lt; |nodes| (nodes[i].hashCode)
256         *     </td>
257         *     <td valign=top>
258         *     </td>
259         *   </tr>
260         * </table>
261         *
262         * @return the hash code for this <b>Predicate</b>
263         * @see
264         *   <a href=
265         *      "../../srchtml/project3/Predicate.html#hashCode"
266         *      >Source Code
267         *   </a>
268         */
269        public int hashCode()
270        {
271            long sum = 0;
272            Iterator<Node> iter = nodes.iterator();
273            while(iter.hasNext()){
274                sum += (long)((Argument)iter.next()).hashCode();
275            };
276            int result = (int)sum;
277            return result;
278        }

426         *   <tr>
427         *     <td valign=top>
428         *       &nbsp;
429         *     </td>
430         *     <td valign=top>
431         *       lex = lex'[1..|lex'|-1]
432         *     </td>
433         *     <td valign=top>
434         *     </td>
435         *   </tr>
436         * </table>
437         *
438         * @throws ParserException
439         *
440         * @param lex the lexical analyzer we will get an <b>Argument</b> from.
441         *
442         * @return the <b>Argument</b> we just created.
443         * @see
444         *   <a href=
445         *      "../../srchtml/project3/Predicate.html#createNode"
446         *      >Source Code
447         *   </a>
448         */
449        protected Node createNode(Lex lex)
450            throws ParserException
451        {
452            if(!(lex.front() instanceof Argument))throw new ParserException(
453                "ERROR in Predicate::createNode(Lex)\n" +
454                "   the lex.front() must be either a identifier or a constant");
455            
456            Node result = lex.front();
457            lex.popFront();
458            return result;
459        }

462         * Checks to see if lex.front() is an <i>Identifier</i> or <i>Constant</i>.

464         * <p>
465         * <table>
466         *   <tr>
467         *     <td valign=top>

472         *     </td>
473         *     <td valign=top>
474         *     </td>
475         *   </tr>
476         *   <tr>
477         *     <td valign=top>
478         *       <b>Postcondition</b>:
479         *     </td>
480         *     <td valign=top>
481         *       result = lex.front() <i>instanceof</i> <b>Argument</b>
482         *     </td>
483         *     <td valign=top>
484         *     </td>
485         *   </tr>
486         * </table>
487         *
488         * @param lex the lexical analyzer we will check to see if lex[0] is in
489         *              the first of <b>Argument</b>.
490         *
491         * @return whether lex[0] is in the first of <b>Argument</b>.
492         * @see
493         *   <a href=

496         *   </a>
497         */
498        protected boolean isInFirstOfListElement(Lex lex)
499        {
500            assert lex != null;
501    
502            return lex.front() instanceof Argument;
503        }
