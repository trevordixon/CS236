001    package project3.three;
002    
003    import java.util.HashSet;
004    import java.util.Iterator;
005    
006    import project3.lex.Constant;
007    
008    /**
009     * For every variable in a predicate list we need to store the locations of
010     * where that variable appears (there can be more than one).
011     *
012     * <pre>
013     * Domain Definition:
014     *     locations : Set&lt;<b>Location</b>&gt; -- the locations where a variable appears
015     *
016     * Instance Invariant:
017     *     locations &ne; null
018     * </pre>
019     *
020     * Last Modified: 21 Feb 2006
021     *
022     * @author Scott N. Woodfield
023     * @version 1.0
024     * @see
025     *   <a href=
026     *      "../../srchtml/project3/Locations.html"
027     *      >Source Code
028     *   </a>
029     */
030    public class Locations
031    {
032    //Domain Implementation
033        /**
034         * The set of locations of a variable.  Implements locations in the
035         * domain definition.
036         * @see
037         *   <a href=
038         *      "../../srchtml/project3/Locations.html#locations"
039         *      >Source Code
040         *   </a>
041         */
042        protected HashSet<Location> locations = null;
043    
044    //Constructors
045        /**
046         * The default constructor for <b>Locations</b>.  
047         * 
048         * <p>
049         * <table>
050         *   <tr>
051         *     <td valign=top>
052         *       <b>Precondition</b>:
053         *     </td>
054         *     <td valign=top>
055         *       none
056         *     </td>
057         *     <td valign=top>
058         *     </td>
059         *   </tr>
060         *   <tr>
061         *     <td valign=top>
062         *       <b>Postcondition</b>:
063         *     </td>
064         *     <td valign=top>
065         *       locations = new HashSet<Location>()
066         *     </td>
067         *     <td valign=top>
068         *       AND
069         *     </td>
070         *   </tr>
071         *   <tr>
072         *     <td valign=top>
073         *       &nbsp;
074         *     </td>
075         *     <td valign=top>
076         *       |locations| = 0 
077         *     </td>
078         *     <td valign=top>
079         *     </td>
080         *   </tr>
081         * </table>
082         * @see
083         *   <a href=
084         *      "../../srchtml/project3/Locations.html#DefaultConstructor"
085         *      >Source Code
086         *   </a>
087         */
088        public Locations()
089        {
090            locations = new HashSet<Location>();
091        }
092    
093        /**
094         * The copy constructor.
095         *
096         * <p>
097         * <table>
098         *   <tr>
099         *     <td valign=top>
100         *       <b>Precondition</b>:
101         *     </td>
102         *     <td valign=top>
103         *       none
104         *     </td>
105         *     <td valign=top>
106         *     </td>
107         *   </tr>
108         *   <tr>
109         *     <td valign=top>
110         *       <b>Postcondition</b>:
111         *     </td>
112         *     <td valign=top>
113         *       locs = null &rArr; locations = new HashSet() 
114         *     </td>
115         *     <td valign=top>
116         *       AND
117         *     </td>
118         *   </tr>
119         *   <tr>
120         *     <td valign=top>
121         *       &nbsp;
122         *     </td>
123         *     <td valign=top>
124         *       locs &ne; null &rArr; locations = new HashSet(locs.locations)
125         *     </td>
126         *     <td valign=top>
127         *     </td>
128         *   </tr>
129         * </table>
130         *
131         * @param locs the <b>Locations</b> we are going to make a copy of.
132         * @see
133         *   <a href=
134         *      "../../srchtml/project3/Locations.html#CopyConstructor"
135         *      >Source Code
136         *   </a>
137         */
138        public Locations(Locations locs)
139        {
140            if(locs == null){
141                locations = new HashSet<Location>();
142            } else {
143                locations = new HashSet<Location>(locs.locations);
144            };
145        }
146    
147    //Queries
148        /**
149         * Returns an iterator over locations.
150         * 
151         * <p>
152         * <table>
153         *   <tr>
154         *     <td valign=top>
155         *       <b>Precondition</b>:
156         *     </td>
157         *     <td valign=top>
158         *       none
159         *     </td>
160         *     <td valign=top>
161         *     </td>
162         *   </tr>
163         *   <tr>
164         *     <td valign=top>
165         *       <b>Postcondition</b>:
166         *     </td>
167         *     <td valign=top>
168         *       result = locations.iterator()
169         *     </td>
170         *     <td valign=top>
171         *     </td>
172         *   </tr>
173         * </table>
174         *
175         * @return an iterator over all of the <b>Location</b>s of this
176         *         <b>Locations</b>.
177         * @see
178         *   <a href=
179         *      "../../srchtml/project3/Locations.html#iterator"
180         *      >Source Code
181         *   </a>
182         */
183        public Iterator<Location> iterator()
184        {
185            return locations.iterator();
186        }
187    
188    //Commands
189        /**
190         * Adds a location to locations.
191         * 
192         * <p>
193         * <table>
194         *   <tr>
195         *     <td valign=top>
196         *       <b>Precondition</b>:
197         *     </td>
198         *     <td valign=top>
199         *       location &ne; null
200         *     </td>
201         *     <td valign=top>
202         *     </td>
203         *   </tr>
204         *   <tr>
205         *     <td valign=top>
206         *       <b>Postcondition</b>:
207         *     </td>
208         *     <td valign=top>
209         *       locations.contains(location)
210         *     </td>
211         *     <td valign=top>
212         *     </td>
213         *   </tr>
214         * </table>
215         *
216         * @param location the location we are going to add to this
217         *                   <b>Locations</b>
218         * @see
219         *   <a href=
220         *      "../../srchtml/project3/Locations.html#addLocation"
221         *      >Source Code
222         *   </a>
223         */
224        public void addLocation(Location location)
225        {
226            assert location != null;
227    
228            locations.add(location);
229        }
230    }
