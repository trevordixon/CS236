007    /**
008     * Logically Lex is a sequence of tokens, called "tokens", 
009     * where we can look only at the front token or "tokens[0]".
010     *
011     * <p>
012     * Syntax:
013     * <table>
014     *   <tr>
015     *     <td><b>Lex</b></td>
016     *     <td>&rarr;</td>
017     *     <td><b>Token</b>+</td>
018     *   </tr>
019     *   <tr>
020     *     <td><b>Token</b></td>
021     *     <td>&rarr;</td>
022     *     <td><i>Identifier</i>+ | <i>Constant</i> | <b>Keyword</b> |
023     *         <b>Symbol</b> | <b>Whitespace</b> | <i>EOF</i> | <b>Undefined</b>
024     *     </td>
025     *   </tr>
026     *   <tr>
027     *     <td><i>Identifier</i></td>
028     *     <td>&rarr;</td>
029     *     <td><b>Letter</b> [ <b>Letter</b> | <b>Digit</b> ]* </td>
030     *   </tr>
031     *   <tr>
032     *     <td><i>Constant</i></td>
033     *     <td>&rarr;</td>
034     *     <td>"'" [<b>Non-QuoteCharacter</b>*
035     *             [["'""'"]+ <b>Non-QuoteCharacter</b>*]* "'"
036     *     </td>
037     *   </tr>
038     *   <tr>
039     *     <td><b>Keyword</b></td>
040     *     <td>&rarr;</td>
041     *     <td>"Schemes" | "Facts" | "Rules" | "Queries"</td>
042     *   </tr>
043     *   <tr>
044     *     <td><b>Symbol</b></td>
045     *     <td>&rarr;</td>
046     *     <td>":" | ":-" | "(" | ")" | "," | "." | "?" </td>
047     *   </tr>
048     *   <tr>
049     *     <td valign=top><b>Whitespace</b></td>
050     *     <td valign=top>&rarr;</td>
051     *     <td>Horizontal_Tab | Line_Feed | Vertical_Tab | Form_Feed | <br>
052     *         Carriage_Return | File_Separator | Group_Separator | <br>
053     *         Record_Separator | Unit_Separator
054     *     </td>
055     *   </tr>
056     *   <tr>
057     *     <td valign=top><b>Undefined</b></td>
058     *     <td valign=top>&rarr;</td>
059     *     <td><i>Any character not defined above</i> | <br>
060     *         <i>A "-" without a previous ":"</i> | <br>
061     *         "'" [<b>Non-QuoteCharacter</b>*
062     *             [["'""'"]+ <b>Non-QuoteCharacter</b>*]* <i>EOF</i> 
063     *     </td>
064     *   </tr>
065     * </table>
066     * <pre>
067     * Domain Definition:
068     *     tokens : Sequence&lt;Token&gt;
069     *
070     * Instance Invariant:
071     *     |tokens| &gt;= 1 AND
072     *     tokens[|tokens|-1].getTokenType() = TokenType.EOF
073     * </pre>
074     *
075     * Last Modified: 10 Jan 2006
076     *
077     * @author Scott N. Woodfield
078     * @version 1.0
079     * @see <a href="../../srchtml/lex/Lex.html">Source Code</a>
080     */

262        /**
263         * Logically this removes the front token (<b>tokens</b>[0]) from
264         * <b>tokens</b>. 
265         *
266         * <p>
267         * From an implementation point of view, we look at file.front() to
268         * determine which kind of token we are going to create next and then
269         * create it.  Notice that unlike project 1, we ignore whitespace.  Also,
270         * if we find illegal characters, we throw an excpetion rather than return
271         * a token.  Also, there are several tokens which were legal in projects
272         * 1 and 2 but are now illegal.  See the definition of
273         * {@link TokenType TokenType} for a list of these token types.
274         *
275         * <p>
276         * <font size = 1>
277         * <table>
278         *   <tr>
279         *     <td valign=top>
280         *       <font size=3><b>Precondition</b></font>:
281         *     </td>
282         *     <td valign=bottom>
283         *       After skipping whitespace, we are at the beginning of a valid
284         *       token.
285         *     </td>
286         *     <td valign=bottom>
287         *     </td>
288         *   </tr>
289         *   <tr><td>&nbsp</td></tr>
290         *   <tr>
291         *     <td valign=top>
292         *       <font size = 3><b>Logical Postcondition</b></font>:
293         *     </td>
294         *     <td valign=bottom>
295         *       |<i>tokens</i>|  = 1 &rArr;
296         *         (<i>tokens</i> = <i>tokens'</i> AND
297         *          <i>tokens</i>[0].getTokenType() = TokenType.EOF)<br/>
298         *       |<i>tokens</i>| &ne; 1 &rArr; <i>tokens</i> =
299         *         <i>tokens'</i>[1..|<i>tokens'</i>|-1]
300         *     </td>
301         *     <td valign=top>
302         *       AND </br>
303         *       <br/>
304         *     </td>
305         *   </tr>
306         *   <tr><td>&nbsp</td></tr>
307         *   <tr>
308         *     <td valign=middle>
309         *       <font size=3><b>Implementation Postcondition</b></font>:
310         *     </td>
311         *     <td valign=bottom>
312         *       file.front() = "'" &rArr; token = getConstant()              <br/>
313         *     </td>
314         *     <td valign=bottom>
315         *       AND
316         *     </td>
317         *   </tr>
318         *   <tr>
319         *     <td>
320         *       &nbsp;
321         *     </td>
322         *     <td valign=top>
323         *     isLetter(file.front()) &rArr; token = getIdentifierOrKeyword() <br/>
324         *     </td>
325         *     <td valign=top>
326         *       AND
327         *     </td>
328         *   </tr>
329         *   <tr>
330         *     <td>
331         *       &nbsp;
332         *     </td>
333         *     <td valign=top>
334         *     file.front() = ',' &rArr; token =
335         *         (new (TokenType.COMMA, ",", File'.getLineNumber()) AND
336         *          file = file'.popFront())                                  <br/>
337         *     </td>
338         *     <td valign=top>
339         *       AND
340         *     </td>
341         *   </tr>
342         *   <tr>
343         *     <td>
344         *       &nbsp;
345         *     </td>
346         *     <td valign=top>
347         *     file.front() = '.' &rArr; token =
348         *         (new (TokenType.PERIOD, ".", File'.getLineNumber()) AND
349         *          file = file'.popFront())                                  <br/>
350         *     </td>
351         *     <td valign=top>
352         *       AND
353         *     </td>
354         *   </tr>
355         *   <tr>
356         *     <td>
357         *       &nbsp;
358         *     </td>
359         *     <td valign=top>
360         *     file.front() = '(' &rArr; token =
361         *         (new (TokenType.LEFT_PAREN, "(", File'.getLineNumber()) AND
362         *          file = file'.popFront())                                  <br/>
363         *     </td>
364         *     <td valign=top>
365         *       AND
366         *     </td>
367         *   </tr>
368         *   <tr>
369         *     <td>
370         *       &nbsp;
371         *     </td>
372         *     <td valign=top>
373         *     file.front() = ')' &rArr; token =
374         *         (new (TokenType.RIGHT_PAREN, ")", File'.getLineNumber()) AND
375         *          file = file'.popFront())                                  <br/>
376         *     </td>
377         *     <td valign=top>
378         *       AND
379         *     </td>
380         *   </tr>
381         *   <tr>
382         *     <td>
383         *       &nbsp;
384         *     </td>
385         *     <td valign=top>
386         *     file.front() = '?' &rArr; token =
387         *         (new (TokenType.Q_MARK, "?", File'.getLineNumber()) AND
388         *          file = file'.popFront())                                  <br/>
389         *     </td>
390         *     <td valign=top>
391         *       AND
392         *     </td>
393         *   </tr>
394         *   <tr>
395         *     <td>
396         *       &nbsp;
397         *     </td>
398         *     <td valign=top>
399         *     file.front() = ':' &rArr; 
400         *         (file[1] = '-' &rArr; token = 
401         *            &nbsp;&nbsp;&nbsp;
402         *            (new (TokenType.COLON_DASH, ":-", File'.getLineNumber()) AND
403         *             file = file'.popFront().popFront()) <br/>
404         *          file[1] &ne; '-' &rArr; token = 
405         *            &nbsp;&nbsp;&nbsp;
406         *            (new (TokenType.COLON, ":", File'.getLineNumber()) AND
407         *             file = file'.popFront()) 
408         *         )                                                          <br/>
409         *     </td>
410         *     <td valign=top>
411         *       AND
412         *     </td>
413         *   </tr>
414         *   <tr>
415         *     <td>
416         *       &nbsp;
417         *     </td>
418         *     <td valign=top>
419         *     file.front() = File.EOF &rArr;
420         *         new (TokenType.EOF, "", File'.getLineNumber())
421         *     </td>
422         *     <td valign=top>
423         *       AND
424         *     </td>
425         *   </tr>
426         * </table>
427         * </font> 
428         *
429         * @throws ParserException When we find an invalid character or
430         *                         unterminated string.
431         * @see <a href="../../srchtml/lex/Lex.html#popFront">Source Code</a>
432         */
433        public void popFront()
434            throws ParserException
435        {
436            skipWhiteSpace();
437            if(file.front() == file.EOF){  
438                token = new Token(TokenType.EOF, "", file.getLineNumber());
439            } else if(file.front() == '\''){
440                token = getConstant();
441            } else if(Character.isLetter(file.front())){
442                token = getIdentifierOrKeyword();
443            } else if(file.front() == ':'){ //process ":-" or ":'
444                int line = file.getLineNumber();
445                file.popFront();
446                if(file.front() == '-'){
447                    file.popFront();
448                    token = new Token(TokenType.COLON_DASH, ":-", line);
449                } else {
450                    token = new Token(TokenType.COLON, ":", line);
451                };
452            } else { //process single character tokens
453                int line = file.getLineNumber();
454                switch(file.front()){
455                    case ',': token = new Token(TokenType.COMMA, ",", line);
456                              break;
457                    case '.': token = new Token(TokenType.PERIOD, ".", line);
458                              break;
459                    case '(': token = new Token(TokenType.LEFT_PAREN,"(",line);
460                              break;
461                    case ')': token = new Token(TokenType.RIGHT_PAREN,")",
462                                                 line);
463                              break;
464                    case '?': token = new Token(TokenType.Q_MARK, "?", line);
465                              break;
466                    default:  throw new ParserException(
467                                 "Error on line " + line + ".\n" +
468                                  "Invalid character, '" + file.front() + 
469                                  "', found.");
470                };
471                file.popFront();
472            };
473        }

518        /**
519         * Gets a String Token.  A string token begins with a "'" followed by
520         * zero or more characters and is terminated by a matching "'".  A "'"
521         * can show up inside a string if it is represented as a double single 
522         * quote "''".
523         *
524         * <p>
525         * <table>
526         *   <tr>
527         *     <td valign=top>
528         *       <b>Precondition</b>:
529         *     </td>
530         *     <td valign=top>
531         *       <i>file</i>.front() = '\'' <br/>
532         *       this is the beginning of a terminated string.
533         *     </td>
534         *     <td valign=top>
535         *       AND <br/>
536         *       &nbsp;
537         *     </td>
538         *   </tr>
539         *   <tr>
540         *     <td valign=top>
541         *       <b>Postcondition</b>:
542         *     </td>
543         *     <td valign=top>
544         *       result = new Token(TokenType.STRING,
545         *                                   <i>file</i>'[1..n-1],
546         *                                   <i>file</i>'.getLineNumber())  <br/>
547         *                <i>file</i> = <i>file</i>'[n+1..|<i>file</i>'-1|] <br/>
548         *                <i>file</i>'[n] = "'"                             <br/>
549         *                Every "'" in <i>file</i>'[1..n-1] must be part of a
550         *                double quote
551         *     </td>
552         *     <td valign=top>
553         *       AND <br/>
554         *       AND <br/>
555         *       AND <br/>
556         *       &nbsp;
557         *     </td>
558         *   </tr>
559         * </table>
560         *
561         * @throws ParserException -- When we find an unterminated String.
562         *
563         * @return the token containing the string found
564         * @see <a href="../../srchtml/lex/Lex.html#getConstant">Source Code</a>
565         */
566        protected Constant getConstant()
567            throws ParserException
568        {
569            //PreCondition: file.front() == '\''
570            int line = file.getLineNumber();
571            if(file.front() != '\'')
572                throw new ParserException("Error on line " + line + ".\n" +
573                                    "A string must start with a '.");
574    
575    
576            file.popFront(); // throw away the leading quote
577    
578            StringBuffer result = new StringBuffer();
579            boolean foundEnd = false;
580            while(!foundEnd && file.front() != file.EOF){
581                if(file.front() == '\''){
582                    file.popFront();
583                    if(file.front() == '\''){  //found a legal double quote
584                        result.append('\'');
585                        file.popFront();
586                    } else {
587                        foundEnd = true;       //found the end of the quote
588                    };
589                } else {
590                    result.append(file.front());
591                    file.popFront();
592                };
593            };
594           
595            Constant resultToken = null;
596            if(foundEnd){
597                resultToken = new Constant(result.toString(),line);
598            } else {
599                throw new ParserException("Error on line " + line + ".\n" +
600                                    "Unterminated string found");
601            };
602            return resultToken;
603        }

610         * @return the identifier or keyword found
611         *
612         * <p>
613         * <table>
614         *   <tr>
615         *     <td valign=top>
616         *       <b>Precondition</b>:
617         *     </td>
618         *     <td valign=top>

620         *     </td>
621         *     <td valign=top>

623         *   </tr>

627         *     </td>

722        protected Token getIdentifierOrKeyword()
723            throws ParserException
724        {
725    
726            //The string may cover many lines, but, the lineNumber of the token
727            //is where the token starts. So, we get the lineNumber before we do
728            //anything else.
729            int line = file.getLineNumber();
730    
731            StringBuffer result = new StringBuffer();
732            while(Character.isLetterOrDigit(file.front())){
733                result.append(file.front());
734                file.popFront();
735            };
736                
737            String value = result.toString(); 
738    
739            //This checks to see if the identifier is actually a key word.:wq
740            Token resultToken = null;
741            if(value.equalsIgnoreCase("Schemes")){
742                resultToken = new Token(TokenType.SCHEMES, value, line);
743            } else if(value.equalsIgnoreCase("Facts")){
744                resultToken = new Token(TokenType.FACTS, value, line);
745            } else if(value.equalsIgnoreCase("Rules")){
746                resultToken = new Token(TokenType.RULES, value, line);
747            } else if(value.equalsIgnoreCase("Queries")){
748                resultToken = new Token(TokenType.QUERIES, value, line);
749            } else {
750                resultToken = new Identifier(value, line);
751            };
752    
753            return resultToken;
754        }
