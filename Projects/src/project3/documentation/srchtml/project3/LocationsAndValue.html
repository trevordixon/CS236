package project3;

import java.util.HashSet;
import java.util.Iterator;

import lex.Constant;


public class LocationsAndValue
    extends Locations
{
//Domain Implementation
//  HashSet locations -- inherited from Locations

    
    protected Constant value = null;
    

//Constructors
    
    public LocationsAndValue()
    {
        super();
    }

    
    public LocationsAndValue(LocationsAndValue locs)
    {
        super(locs);
        value = locs.value;
    }


//Queries
    
    public Constant getValue()
    {
        return value;
    }

//Commands
    
    public void setValue(Constant value)
    {
        assert value != null;

        this.value = value;
    }
}
