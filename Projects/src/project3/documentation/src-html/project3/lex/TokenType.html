001    package project3.lex;

003    /**
004     * An enumeration of the different types of tokens.  Assumes you will use
005     * version 1.5 (or higher) of the java compiler.
006     *
007     * <p>
008     * The token types: LT, LE, GT, GE, EQ, NE, MULTIPLY, ADD, COMPARATOR, 
009     *                  EXPRESSION, and WHITESPACE 
010     *
011     * <br/>
012     * were needed for projects 1 and 2 but are not in the grammar for project 3.
013     *
014     * <p>
015     * Domain:<br/>
016     * &nbsp;&nbsp;{<br/>
017     *     &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;
018     *     COLON, COLON_DASH, COMMA, EOF, FACTS, ID, LEFT_PAREN, <br/>
019     *     &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;
020     *     PERIOD, Q_MARK, QUERIES, RIGHT_PAREN, RULES, SCHEMES, STRING <br/>
021     * &nbsp;&nbsp;}
022     *
023     * <p>
024     * Last Modified: 10 Jan 2006
025     *
026     * @author Scott N. Woodfield
027     * @version 1.0
028     * @see <a href="../../srchtml/lex/TokenType.html">Source Code</a>
029     */
030    public enum TokenType
031    {
032        EOF, ID, SCHEMES, FACTS, RULES, QUERIES, STRING, COLON,
033        COLON_DASH, COMMA, LEFT_PAREN, PERIOD, Q_MARK, RIGHT_PAREN, UNDEFINED
034    }
