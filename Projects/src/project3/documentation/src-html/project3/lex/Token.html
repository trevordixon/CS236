006    /**
007     * A lexical analyzer produces a list of tokens.  This defines a token.
008     * A Token consists of a type, value, and lineNumber.  We inherit from
009     * <b>project3.Node</b> so that we can add tokens to constant lists,
010     * attribute lists, or argument lists.
011     *
012     * <p>
013     * <pre>
014     * Domain:
015     *      type       : TokenType
016     *      value      : String
017     *      lineNumber : int
018     *
019     * Instance Invariant:
020     *      type &ne; null AND
021     *      value &ne; null AND
022     *      type = TokenType.EOF &rArr; value = ""
023     * </pre>
024     *
025     * Last Modified: 2 Mar 2006
026     * @author Scott N. Woodfield
027     * @version 1.0
028     * @see <a href="../../srchtml/lex/Token.html">Source Code</a>
029     */
030    public class Token 
031        extends Node
032    {

034    
035        /**
036         * The COMMA Token.  Used when we need a comma token but don't care about
037         * the line number.  This token is currently used in
038         * {@link project3.three.NamedList#getSeparator()
039         *      project3.three.NamedList.getSeparator()}
040         * and in
041         * {@link project3.PredicateList#getSeparator()
042         *      project3.PredeicateList.getSeparator()}. 
043         * @see <a href="../../srchtml/lex/Token.html#COMMA">Source Code</a>
044         */
045        public static final Token COMMA = new Token(TokenType.COMMA, ",");

050         * after the token has been created.
051         * @see <a href="../../srchtml/lex/Token.html#type">Source Code</a>
052         */
053        protected TokenType type  = TokenType.UNDEFINED;
054    
055        /**
056         * The value of the token.  The sequence of characters from the input stream
057         * that were parsed to create this token.  Some tokens with the same
058         * token type will always have the same value.  For instance, tokens of
059         * type TokenType.SEMICOLON will always have the value ";".  Other tokens,
060         * such as strings and identifiers can have different values.  Tokens of
061         * type EOF will have the "" or empty string value.
062         * @see <a href="../../srchtml/lex/Token.html#value">Source Code</a>
063         */
064        protected String    value = null;

067         * The line on which the first character of the token was found.

069         */
070        protected int    lineNumber = -1;

073        /**
074         * Constructs a token from a type and value.  Used primarily to create
075         * constants.
076         *
077         * <p>
078         * <table>
079         *   <tr>
080         *     <td> <b>Precondition</b>: </td>
081         *     <td> None</td>
082         *   </tr>
083         *   <tr>
084         *     <td valign=top>
085         *       <b>PostCondition</b>:
086         *     </td>
087         *     <td valign=top> 
088         *       this.type = type<br/>
089         *       this.type = TokenType.EOF &rArr; this.value = ""<br/>
090         *       this.type &ne; TokenType.EOF &rArr; this.value = value<br/>
091         *       this.lineNumber = -1
092         *     </td>
093         *     <td valign=top>
094         *       AND <br/>
095         *       AND <br/>
096         *       AND <br/>
097         *       &nbsp;
098         *     </td>
099         *   </tr>
100         * </table>
101         * @param type the type of the token.  
102         * @param value the value of the token.
103         * @see <a href="../../srchtml/lex/Token.html#StdConstructor">Source Code</a>
104         */
105        private Token(TokenType type, String value)
106        {
107            this.type = type;
108            if(type == TokenType.EOF){
109               this.value = "";
110            } else {
111               this.value = value;
112            };
113    
114            this.lineNumber = -1;
115        }

117        /**
118         * Constructs a token from a type, value, and lineNumber.
119         *
120         * <p>
121         * <table>
122         *   <tr>
123         *     <td valign=top>
124         *         <b>Precondition</b>:
125         *     </td>
126         *     <td valign=top>
127         *       checkConstructorPrecondition()
128         *     </td>
129         *     <td>
130         *     </td>
131         *   </tr>
132         *   <tr>
133         *     <td valign=top>
134         *         <b>PostCondition</b>:
135         *     </td>
136         *     <td valign=top>
137         *       this.type = type <br/>
138         *       this.type = TokenType.EOF &rArr; this.value = "" <br/>
139         *       this.type &ne; TokenType.EOF &rArr; this.value = value <br/>
140         *       this.lineNumber = lineNumber
141         *     </td>
142         *     <td valign=top>
143         *       AND<br/>
144         *       AND<br/>
145         *       AND<br/>
146         *       &nbsp;
147         *     </td>
148         *   </tr>
149         * </table>
150         *
151         * @param type the type of the token.  
152         * @param value the value of the token.
153         * @param lineNumber the lineNumber of the token.
154         *
155         * @throws ParserException
156         * @see <a href="../../srchtml/lex/Token.html#StdConstructor2">Source Code</a>
157         */
158        public Token(TokenType type, String value, int lineNumber)
159            throws ParserException 
160        {
161            checkConstructorPrecondition(type, value, lineNumber);
162    
163            this.type = type;
164            if(type == TokenType.EOF){
165               this.value = "";
166            } else {
167               this.value = value;
168            };
169    
170            this.lineNumber = lineNumber;
171        }

218        /**
219         *Compares two Tokens to see if they are equal. Notice that we only use
220         * the type and value to determine if two tokens are equal. We ignore the
221         * line number.
222         *
223         * <p>
224         * <table>
225         *   <tr>
226         *     <td valign=top>
227         *       <b>Precondition</b>:
228         *     </td>
229         *     <td valign=top>
230         *       none
231         *     </td>
232         *     <td valign=top>
233         *     </td>
234         *   </tr>
235         *   <tr>
236         *     <td valign=top>
237         *       <b>PostCondition</b>:
238         *     </td>
239         *     <td valign=top>
240         *       result = object &ne; null <br/>
241         *       object <i>instanceof</i> Token <br/>
242         *       type = object.type <br/>
243         *       value = object.value 
244         *     </td>
245         *     <td valign=top>
246         *       AND<br/>
247         *       AND<br/>
248         *       AND<br/>
249         *       &nbsp;
250         *     </td>
251         *   </tr>
252         * </table>
253         *
254         * @param object the object we are comparing this to
255         *
256         * @return true iff the objects are equal
257         * @see <a href="../../srchtml/lex/Token.html#equals">Source Code</a>
258         */
259        public boolean equals(Object object)
260        {
261            boolean result = object != null && object instanceof Token;
262            if(result){
263                Token token = (Token)object;
264                result = type == token.type &&
265                         ((value ==  null && token.value == null) ||
266                          (value != null && value.equals(token.value))
267                         );
268            };
269            return result;
270        }

465        /**
466         * Checks the precondition of the non-copy constructors.  Since we only
467         * check preconditions, the only purpose of this method is to throw an
468         * exception if an error is found.  Otherwise it does nothing.
469         * 
470         * <p>
471         * <table>
472         *   <tr>
473         *     <td valign=top>
474         *       <b>Precondition</b>:
475         *     </td>
476         *     <td valign=top>
477         *       type &ne; null <br/>
478         *       lineNumber &gt;= 1 <br/>
479         *       type &ne; TokenType.EOF &rArr; value &ne; null <br/>
480         *       type &ne; TokenType.EOF AND type &ne; TokenType.STRING &rArr;
481         *            |value| &gt; 0 <br/>
482         *       type == TokenType.STRING &rArr; (isLetter(value[0]) <br/>
483         *       &forall;<sub>i</sub> (0 &lt;= i AND i &lt; |value| &rArr;
484         *           (isLetterOrDigit(value[i]))) <br/>
485         *       type == TokenType.SCHEMES &rArr; value.equalsIgnoreCase(
486         *           "Schemes") <br/>
487         *       type == TokenType.Facts &rArr; value.equalsIgnoreCase("Facts")<br/>
488         *       type == TokenType.Rules &rArr; value.equalsIgnoreCase("Rules")<br/>
489         *       type == TokenType.Rules &rArr; value.equalsIgnoreCase("Queries")
490         *           <br/>
491         *       type == TokenType.COLON_DASH &rArr; value = ":-" <br/>
492         *       type == TokenType.COLON      &rArr; value = ":"  <br/>
493         *       type == TokenType.COMMA      &rArr; value = ","  <br/>
494         *       type == TokenType.PERIOD     &rArr; value = "."  <br/>
495         *       type == TokenType.LEFT_PAREN &rArr; value = "(" <br/>
496         *       type == TokenType.RIGHT_PAREN&rArr; value = ")" <br/>
497         *       type == TokenType.Q_MARK     &rArr; value = "?" <br/>
498         *       lineNumber &gt;= 1
499         *     </td>
500         *     <td valign=top>
501         *       AND<br/>
502         *       AND<br/>
503         *       AND<br/>
504         *       AND<br/>
505         *       AND<br/>
506         *       AND<br/>
507         *       AND<br/>
508         *       AND<br/>
509         *       AND<br/>
510         *       AND<br/>
511         *       AND<br/>
512         *       AND<br/>
513         *       AND<br/>
514         *       AND<br/>
515         *       AND<br/>
516         *       AND<br/>
517         *       AND<br/>
518         *       &nbsp;
519         *     </td>
520         *   </tr>
521         *   <tr>
522         *     <td valign=top>
523         *       <b>PostCondition</b>:
524         *     </td>
525         *     <td valign=top>
526         *       true
527         *     </td>
528         *     <td valign=top>
529         *     </td>
530         *   </tr>
531         *  
532         * </table>
533         *
534         * @param type  the token type to be checked.
535         * @param value the value of the new token.
536         * @param lineNumber the lineNumber of the new token
537         *
538         * @throws ParserException
539         * @see <a href="../../srchtml/lex/Token.html#constructorPrecond">Source Code</a>
540         */
541        protected void checkConstructorPrecondition(
542            TokenType type, String value, int lineNumber)
543            throws ParserException
544        {
545            String errorMessage = null;
546            if(type == null){
547                errorMessage = "type cannot be null";
548            } else if(type != TokenType.EOF && value == null){ 
549                errorMessage = "type != TokenType.EOF => value != null";
550            } else if(type != TokenType.EOF && type != TokenType.STRING &&
551                      value.length() == 0
552                     ){ 
553                errorMessage = "If the token is not an EOF or String token " +
554                                "then |value| > 0";
555            } else if(type == TokenType.ID &&
556                      !(Character.isLetter(value.charAt(0)) &&
557                        allTheRestMustBeLettersOrDigits(value))){
558                errorMessage = "type = TokenType.STRING =>\n" +
559                                "    the first character in value must be a letter"+
560                                " and\n" +
561                                "    the rest must be letters or digits";
562            } else if(type == TokenType.SCHEMES &&
563                      !value.equalsIgnoreCase("Schemes")){
564                errorMessage = "type = TokenType.SCHEMES =>\n" +
565                                "    value.equalsIgnoreCase(\"Schemes\")";
566            } else if(type == TokenType.FACTS &&
567                      !value.equalsIgnoreCase("Facts")){ 
568                errorMessage = "type = TokenType.FACTS =>\n" +
569                                "    value.equalsIgnoreCase(\"Facts\")";
570            } else if(type == TokenType.RULES &&
571                      !value.equalsIgnoreCase("Rules")){ 
572                errorMessage = "type = TokenType.RULES =>\n" +
573                                "    value.equalsIgnoreCase(\"Rules\")";
574            } else if(type == TokenType.QUERIES &&
575                      !value.equalsIgnoreCase("Queries")){ 
576                errorMessage = "type = TokenType.QUERIES =>\n" +
577                                "    value.equalsIgnoreCase(\"Queries\")";
578            } else if(type == TokenType.COLON_DASH && !value.equals(":-")){
579                errorMessage = "type = TokenType.COLON_DASH => value = \":-\""; 
580            } else if(type == TokenType.COLON && !value.equals(":")){
581                errorMessage = "type = TokenType.COLON => value = \":\""; 
582            } else if(type == TokenType.COMMA && !value.equals(",")){
583                errorMessage = "type = TokenType.COMMA => value = \",\""; 
584            } else if(type == TokenType.PERIOD && !value.equals(".")){
585                errorMessage = "type = TokenType.PERIOD => value = \".\""; 
586            } else if(type == TokenType.LEFT_PAREN && !value.equals("(")){
587                errorMessage = "type = TokenType.LEFT_PAREN => value = \"(\""; 
588            } else if(type == TokenType.RIGHT_PAREN && !value.equals(")")){
589                errorMessage = "type = TokenType.RIGHT_PAREN => value = \")\""; 
590            } else if(type == TokenType.Q_MARK && !value.equals("?")){
591                errorMessage = "type = TokenType.Q_MARK => value = \"?\""; 
592            } else if(lineNumber <= 0){
593                errorMessage = "lineNumber must be >= 1";
594            };
595            if(errorMessage != null)
596                throw new ParserException(
597                   "ERROR in one of the Token::constructor\n" + errorMessage);
598        }

600        /**
601         * Returns true if value[1..|value|-1] are all letters or digits, otherwise
602         * it returns false.
603         *
604         * <p>
605         * <table>
606         *   <tr>
607         *     <td valign=top>
608         *       <b>Precondition</b>:
609         *     </td>
610         *     <td valign=top>
611         *       none
612         *     </td>
613         *     <td valign=top>
614         *     </td>
615         *   </tr>
616         *   <tr>
617         *     <td valign=top>
618         *       <b>Postcondition</b>:
619         *     </td>
620         *     <td valign=top>
621         *       result = value &ne; null <br/>
622         *       &forall;<sub>i</sub> (0 &lt; i AND i &lt; |value| &rArr;
623         *           isLetterOrDigit(str[i]))
624         *     </td>
625         *     <td valign=top>
626         *       AND <br/>
627         *       &nbsp;
628         *     </td>
629         *   </tr>
630         * </table>
631         *
632         * @param value the string to be inspected.
633         *
634         * @return whether or not all but the first character are letters and
635         *         digits.
636         * @see <a href="../../srchtml/lex/Token.html#lettersOrDigits">Source Code</a>
637         */
638        protected boolean allTheRestMustBeLettersOrDigits(String value)
639        {
640            boolean result = value != null;
641    
642            for(int i = 1; result && i < value.length(); i++){
643                result = Character.isLetterOrDigit(value.charAt(i));
644            };
645    
646            return result;
647        }
