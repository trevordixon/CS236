001    package project3.lex;
002    
003    import project3.three.Node;
004    import project3.three.ParserException;
005    
006    /**
007     * A <i>Constant</i> token. This is not needed for assignment 1, but is
008     * included in the <i>lex</i> package to make it easier to read the code
009     * in project 3.
010     *
011     * <p>
012     * <pre>
013     * Sytnax: "'" <b>Non-QuoteCharacter</b>* [["'" "'"]+
014     *             <b>Non-QuoteCharacter</b>]* "'"
015     * </pre>
016     *
017     * <p>
018     * <pre>
019     * Domain Definition:
020     *     type       : TokenType
021     *     value      : String
022     *     lineNumber : int
023     *
024     * Instance Invariant:
025     *     type = TokenType.STRING
026     * </pre>
027     *
028     * Last Modified: 10 Jan 2006
029     *
030     * @author Scott N. Woodfield
031     * @version 1.0
032     * @see <a href="../../srchtml/lex/Constant.html">Source Code</a>
033     */
034    public class Constant 
035        extends Argument
036    {
037    //Constants
038    //  TokenType type       -- inherited from Token through Argument
039    //  String    value      -- inherited from Token through Argument
040    //  int       lineNumber -- inherited from Token through Argument
041    
042    //Domain Implementation -- Inherited from Token
043    
044    //Constructors
045        /**
046         * Constructs a <i>Constant</i> from a type, value, and lineNumber
047         *
048         * <p>
049         * <table>
050         *   <tr>
051         *     <td valign=top>
052         *       <b>Precondition</b>:
053         *     </td>
054         *     <td valign=top>
055         *       Token.Precondition
056         *     </td>
057         *     <td valign=top>
058         *     </td>
059         *   </tr>
060         *   <tr>
061         *     <td valign=top>
062         *       <b>Postcondition</b>:
063         *     </td>
064         *     <td valign=top>
065         *       Token.Postcondition
066         *     </td>
067         *     <td valign=top>
068         *     </td>
069         *   </tr>
070         * </table>
071         *
072         * @param value the value of the constant.
073         * @param lineNumber the line number this token was found on.
074         *
075         * @throws ParserException
076         * @see <a href="../../srchtml/lex/Constant.html#StdConstructor">Source Code</a>
077         */
078        public Constant(String value, int lineNumber)
079            throws ParserException
080        {
081            super(TokenType.STRING, value, lineNumber);
082        }
083    
084        /**
085         * Constructs a <i>Constant</i> from a <b>Token</b>.
086         *
087         * <table>
088         * <p>
089         *   <tr>
090         *     <td valign=top>
091         *       <b>Precondition</b>:
092         *     </td>
093         *     <td valign=top>
094         *       token <i>instanceof</i> <b>Constant</b>
095         *     </td>
096         *     <td valign=top>
097         *     </td>
098         *   </tr>
099         *   <tr>
100         *     <td valign=top>
101         *       <b>Postcondition</b>:
102         *     </td>
103         *     <td valign=top>
104         *       this.equals(token)
105         *     </td>
106         *     <td valign=top>
107         *     </td>
108         *   </tr>
109         * </table>
110         *
111         * @param token the constant <b>Token</b> we will copy from.
112         * @see <a href="../../srchtml/lex/Constant.html#CopyConstructor">Source Code</a>
113         */
114        public Constant(Token token)
115            throws ParserException
116        {
117            super(token);
118            assert token instanceof Constant;
119        }
120    
121    //Queries
122        /**
123         * Convert the constant to a string.
124         *
125         * <p>
126         * <table>
127         *   <tr>
128         *     <td valign=top>
129         *       <b>Precondition</b>:
130         *     </td>
131         *     <td valign=top>
132         *       none
133         *     </td>
134         *     <td valign=top>
135         *     </td>
136         *   </tr>
137         *   <tr>
138         *     <td valign=top>
139         *       <b>Postcondition</b>:
140         *     </td>
141         *     <td valign=top>
142         *       result =  "'" + value + "'"
143         *     </td>
144         *     <td valign=top>
145         *     </td>
146         *   </tr>
147         * </table>
148         * @return the lexical representation of the <b>Constant</b>.
149         * @see <a href="../../srchtml/lex/Constant.html#toString">Source Code</a>
150         */
151        public String toString()
152        {
153             return "'" + value + "'";
154        }
155    }
