006    /**
007     * An <b>Argument</b> token.  This is not needed for assignment 1, but is
008     * included in the <i>lex</i> package to make it easier to read the code
009     * in project 3.
010     *
011     * <p>
012     * Syntax : <b>Argument</b> &rarr; <i>Identifier</i> | <i> Constant</i>
013     *
014     * <p>
015     * <pre>
016     * Domain Definition:
017     *     type       : TokenType
018     *     value      : String
019     *     lineNumber : Integer
020     * 
021     * Instance Invariant:
022     *     &forall;<sub>x</sub> (x <i>instanceof</i> Argument &rArr;
023     *                   x <i>instanceof</i> <i>Identifier</i> OR
024     *                   x <i>instanceof</i> <i>Constant</i>
025     *         )
026     * </pre>
027     *
028     * Last Modified: 21 Feb 2006
029     *
030     * @author Scott N. Woodfield
031     * @version 1.0
032     * @see <a href="../../srchtml/lex/Argument.html">Source Code</a>
033     */
034    abstract public class Argument 
035        extends Token
036    {

044         * Constructs an <b>Argument</b> from a type and value.

049         *
050         * <p>
051         * <table>
052         *   <tr>
053         *     <td valign=top>
054         *       <b>Precondition</b>:
055         *     </td>
056         *     <td valign=top>
057         *       super.Precondition

059         *     <td valign=top>
060         *     </td>
061         *   </tr>
062         *   <tr>
063         *     <td valign=top>
064         *       <b>Postcondition</b>:
065         *     </td>
066         *     <td valign=top>
067         *       super.Postcondition
068         *     </td>
069         *     <td valign=top>
070         *     </td>
071         *   </tr>
072         * </table>
073         *
074         * @param type the type of the new argument.
075         * @param value the value of the new argument.
076         * @param lineNumber the lineNumber of the new argument.
077         *

119         * @see

124            super(token);
125        }

131        /**
132         * Creates a specific copy of the argument depending on the type of the
133         * argument.
134         *
135         * <p>
136         * <table>
137         *   <tr>
138         *     <td valign=top>
139         *       <b>Precondition</b>:
140         *     </td>
141         *     <td valign=top>
142         *       argument &ne; null
143         *     </td>
144         *     <td valign=top>
145         *     </td>
146         *   </tr>
147         *   <tr>
148         *     <td valign=top>
149         *       <b>Postcondition</b>:
150         *     </td>
151         *     <td valign=top>
152         *       argument <i>instanceof</i> <b>Identifier</b> &rArr;
153         *           result = new Identifier(argument) <br/>
154         *       argument <i>instanceof</i> <b>Constant</b> &rArr;
155         *           result = new Constant(argument) <br/>
156         *       &not; argument <i>instanceof</i> <b>Constant</b> AND
157         *       &not; argument <i>instanceof</i> <b>Identifier</b> &rArr;
158         *           result = null
159         *     </td>
160         *     <td valign=top>
161         *       AND <br/>
162         *       AND <br/>
163         *       &nbsp;
164         *     </td>
165         *   </tr>
166         * </table>
167         *
168         * @param argument the argument we are to copy.
169         * @return a copy of the input parameter.
170         * @throws ParserException
171         * @see
172         *   <a href="../../srchtml/lex/Argument.html#createCopy">Source Code</a>
173         */
174         static public Argument createCopy(Argument argument)
175             throws ParserException
176         {
177             assert argument != null;
178       
179             Argument result = null;
180             if(argument instanceof Identifier){
181                 result = new Identifier(argument);
182             } else if(argument instanceof Constant){
183                 result = new Constant((Constant)argument);
184             };
185             return result;
186         }
