001    package project3.three;
002    
003    import java.util.ArrayList;
004    import java.util.Iterator;
005    
006    import project3.lex.Constant;
007    import project3.lex.Identifier;

009    /**
010     * For every query we keep of list a solutions that make the query true.
011     * In this case we represent a solution as an array of constants. Thus we have
012     * a sequence of arrays.
013     *
014     * <p>
015     * <pre>
016     * Domain Definition:
017     *     variables     : Identifier[] 
018     *     setOfSolution : Set<Conststant[]>
019     *
020     * Instance Invariant:
021     *     |variables| &gt; 1 AND
022     *     &forall;<sub>i</sub> (0 &lt;= i AND i &lt; |setOfSolutions| &rArr;
023     *         |setOfSolutions.get(i)| == |variables| AND
024     *          &forall;<sub>j</sub> (&lt;= j AND j &lt; |setOfSolutions.get(i)| &rArr;
025     *              setOfSolutions.get(i).get(j) &ne; null))
026     * </pre>
027     *
028     * Last Modified: 21 Feb 2006
029     *
030     * @author Scott N. Woodfield
031     * @version 1.0
032     * @see
033     *   <a href=
034     *      "../../srchtml/project3/SetOfSolutions.html"
035     *      >Source Code
036     *   </a>
037     */
038    public class SetOfSolutions
039    {
040    //Domain Implementation
041        /**
042         * Implementation of variables.
043         * @see
044         *   <a href=
045         *      "../../srchtml/project3/SetOfSolutions.html#variables"
046         *      >Source Code
047         *   </a>
048         */
049        protected Identifier[] variables = null;
050    
051        /**
052         * Implementation of setOfSolutions.
053         * @see
054         *   <a href=
055         *      "../../srchtml/project3/SetOfSolutions.html#setOfSolutions"
056         *      >Source Code
057         *   </a>
058         */
059        protected ArrayList<Constant[]> setOfSolutions = null;
060    
061    //Constructor
062        /**
063         * Create an empty <b>SetOfSolutions</b>
064         *
065         * <p>
066         * <table>
067         *   <tr>
068         *     <td valign=top>
069         *       <b>Precondition</b>:
070         *     </td>
071         *     <td valign=top>
072         *       variables &ne; null 
073         *     </td>
074         *     <td valign=top>
075         *       AND
076         *     </td>
077         *   </tr>
078         *   <tr>
079         *     <td valign=top>
080         *       &nbsp;
081         *     </td>
082         *     <td valign=top>
083         *       &forall;<sub>i</sub> (0 &lt;= i AND i &lt; |variables| &rArr;
084         *           (variables[i] &ne; null)
085         *     </td>
086         *     <td valign=top>
087         *     </td>
088         *   </tr>
089         *   <tr>
090         *     <td valign=top>
091         *       <b>Postcondition</b>:
092         *     </td>
093         *     <td valign=top>
094         *       this.variables = variables 
095         *     </td>
096         *     <td valign=top>
097         *       AND
098         *     </td>
099         *   </tr>
100         *   <tr>
101         *     <td valign=top>
102         *       &nbsp;
103         *     </td>
104         *     <td valign=top>
105         *       setOfSolutions = new ArrayList&lt;<i>Constant</i>[]&gt;()
106         *     </td>
107         *     <td valign=top>
108         *     </td>
109         *   </tr>
110         * </table>
111         *
112         * @param variables the list of variables from which we will create a
113         *                    <b>SetOfSolutions</b>
114         * @see
115         *   <a href=
116         *      "../../srchtml/project3/SetOfSolutions.html#PrimaryConstructor"
117         *      >Source Code
118         *   </a>
119         */
120        public SetOfSolutions(Identifier[] variables)
121        {
122            assert variables != null;
123    
124            boolean foundNull = false;
125            for(int i = 0; i < variables.length && !foundNull; i++){
126                foundNull = variables[i] == null;
127            };
128            assert !foundNull;
129    
130            this.variables = variables;
131            setOfSolutions = new ArrayList<Constant[]>();
132        }
133    
134    //Query
135        /**
136         * Convert a <b>SetOfSolutions</b> to a string.
137         *
138         * <p>
139         * <table>
140         *   <tr>
141         *     <td valign=top>
142         *       <b>Precondition</b>:
143         *     </td>
144         *     <td valign=top>
145         *       none
146         *     </td>
147         *     <td valign=top>
148         *     </td>
149         *   </tr>
150         *   <tr>
151         *     <td valign=top>
152         *       <b>Postcondition</b>:
153         *     </td>
154         *     <td valign=top>
155         *        |setOfSolutions| = 0 &rArr; result = " No\n"
156         *     </td>
157         *     <td valign=top>
158         *       AND
159         *     </td>
160         *   </tr>
161         *   <tr>
162         *     <td valign=top>
163         *       &nbsp;
164         *     </td>
165         *     <td valign=top>
166         *       |setOfSolutions| &ne; 0 &rArr; 
167         *       result = &sum; 0 &lt;= i &lt; |setOfSolutions|
168         *              (printSolutions(result, setOfSolutions[i]) + "\n"
169         *     </td>
170         *     <td valign=top>
171         *     </td>
172         *   </tr>
173         * </table>
174         *
175         * @return the string representation of a <b>SetOfSolutions</b>
176         * @see
177         *   <a href=
178         *      "../../srchtml/project3/SetOfSolutions.html#toString"
179         *      >Source Code
180         *   </a>
181         */
182        public String toString()
183        {
184            StringBuffer result = new StringBuffer();
185    
186            if(setOfSolutions.size() == 0){
187                result.append(" No\n");
188            } else {
189                result.append(" Yes("+setOfSolutions.size()+")\n");
190                Iterator<Constant[]> solutionsIterator =
191                    setOfSolutions.iterator();
192                while(solutionsIterator.hasNext()){
193                    printSolutions(result, solutionsIterator.next());
194                };
195            };
196            return result.toString();
197        }
198    
199    //Commands
200        /**
201         * Add a solution to the <b>SetOfSolutions</b>.
202         *
203         * Postcondition: setOfSolutions.contains(solution);
204         * <p>
205         * <table>
206         *   <tr>
207         *     <td valign=top>
208         *       <b>Precondition</b>:
209         *     </td>
210         *     <td valign=top>
211         *       solution &ne; null
212         *     </td>
213         *     <td valign=top>
214         *       AND
215         *     </td>
216         *   </tr>
217         *   <tr>
218         *     <td valign=top>
219         *       &nbsp;
220         *     </td>
221         *     <td valign=top>
222         *         |solution| = |variables|
223         *     </td>
224         *     <td valign=top>
225         *     </td>
226         *   </tr>
227         *   <tr>
228         *     <td valign=top>
229         *       <b>Postcondition</b>:
230         *     </td>
231         *     <td valign=top>
232         *       setOfSolutions = setOfSolutions &cup; {solution}
233         *     </td>
234         *     <td valign=top>
235         *     </td>
236         *   </tr>
237         * </table>
238         *
239         * @param solution the solution to be added to the list of solutions.
240         * @see
241         *   <a href=
242         *      "../../srchtml/project3/SetOfSolutions.html#addSolution"
243         *      >Source Code
244         *   </a>
245         */
246        public void addSolution(Constant[] solution)
247        {
248            assert solution != null &&
249                   solution.length == variables.length; 
250    
251            if(!setOfSolutions.contains(solution)){
252                setOfSolutions.add(solution);
253            };
254        }

256    //Auxillary Methods and Constants
257        /**
258         * Convert an array of solutions to a string.
259         *
260         * <p>
261         * <table>
262         *   <tr>
263         *     <td valign=top>
264         *       <b>Precondition</b>:
265         *     </td>
266         *     <td valign=top>
267         *        result &ne; null
268         *     </td>
269         *     <td valign=top>
270         *     </td>
271         *   </tr>
272         *   <tr>
273         *     <td valign=top>
274         *       <b>Postcondition</b>:
275         *     </td>
276         *     <td valign=top>
277         *       |solutions| = 0 &rArr; result = result' + "\n"
278         *     </td>
279         *     <td valign=top>
280         *       AND
281         *     </td>
282         *   </tr>
283         *   <tr>
284         *     <td valign=top>
285         *       &nbsp;
286         *     </td>
287         *     <td valign=top>
288         *       |solutions| &ne; 0 &rArr; <br>
289         *         &nbsp;&nbsp;&nbsp;result = result' +
290         *         &nbsp;&nbsp;&nbsp;variables[0] + " = " + solutions[i] + <br>
291         *         &nbsp;&nbsp;&nbsp;&sum; 1 &lt;= i &lt; |variables|
292         *             (", " + variables[i] + " = " + solutions[i]) + <br>
293         *         &nbsp;&nbsp;&nbsp;"\n"
294         *     </td>
295         *     <td valign=top>
296         *     </td>
297         *   </tr>
298         * </table>
299         *
300         * @param result the string buffer we will store the result in.
301         * @param solution the solution we will print.
302         * @see
303         *   <a href=
304         *      "../../srchtml/project3/SetOfSolutions.html#printSolutions"
305         *      >Source Code
306         *   </a>
307         */
308        protected void printSolutions(StringBuffer result, Constant[] solution)
309        {
310            assert result != null; 
311    
312            if(solution.length > 0){
313                result.append("  " + variables[0] + "=" + solution[0]);
314    
315                int i = 1;        
316                while(i < variables.length){
317                    result.append(", " + variables[i] + "=" + solution[i]);
318                    i++;
319                };
320                result.append('\n');
321            };
322        }
323    
324        /**
325         * Returns whether or not the <b>SetOfSolutions</b> contains the
326         * solution.
327         *
328         * <p>
329         * <table>
330         *   <tr>
331         *     <td valign=top>
332         *       <b>Precondition</b>:
333         *     </td>
334         *     <td valign=top>
335         *       solution &ne; null 
336         *     </td>
337         *     <td valign=top>
338         *       AND
339         *     </td>
340         *   </tr>
341         *   <tr>
342         *     <td valign=top>
343         *       &nbsp;
344         *     </td>
345         *     <td valign=top>
346         *       |solution| = |variables|
347         *     </td>
348         *     <td valign=top>
349         *     </td>
350         *   </tr>
351         *   <tr>
352         *     <td valign=top>
353         *       <b>Postcondition</b>:
354         *     </td>
355         *     <td valign=top>
356         *         result = &exist;<sub>i</sub> (0 &lt;= i AND i &lt;
357         *             |setOfSolutions| &rArr; setOfSolutions[i] = solution)
358         *     </td>
359         *     <td valign=top>
360         *     </td>
361         *   </tr>
362         * </table>
363         *
364         * @param solution we want to check to see if this solution is in
365         *                   this <b>SetOfSolutions</b>
366         *
367         * @return whether the solution is in this <b>SetOfSolutions</b>
368         * @see
369         *   <a href=
370         *      "../../srchtml/project3/SetOfSolutions.html#contains"
371         *      >Source Code
372         *   </a>
373         */
374        private boolean contains(Constant[] solution)
375        {
376            assert solution != null &&
377                   solution.length == variables.length;
378    
379            boolean result = false;
380            Iterator<Constant[]> iter = setOfSolutions.iterator();
381            while(iter.hasNext() && !result){
382                Constant[] constants = iter.next();
383                boolean same = true;
384                for(int i = 0; i < solution.length && same; i++){
385                    same = solution[i].equals(constants[i]);
386                };
387                result = same;
388            };
389    
390            return result;
391        }
