001    package project3.three;
002    
003    import project3.lex.Lex;
004    import project3.lex.TokenType;
005    import project3.lex.Token;
006    
007    /**
008     * A <b>DatalogSegment</b> is an abstract classs that is differentiated from
009     * <b>List</b> by having a name. This class does not override any of the
010     * methods defined in the <b>List</b> class.  Overriding will be done in the
011     * specializations.
012     *
013     * <p>
014     * <pre>
015     * Domain Definition:
016     *     nodes : Sequence&lt;<b>Node</b>&gt; nodes -- inherited from List
017     *     name  : String name
018     * </pre>
019     *
020     * Last Modified: 21 Feb 2006
021     * @author Scott N. Woodfield
022     * @version 1.0
023     * @see
024     *   <a href=
025     *      "../../srchtml/project3/DatalogSegment.html"
026     *      >Source Code
027     *   </a>
028     */
029    abstract public class DatalogSegment
030        extends List
031    {
032    //Domain Implementations
033    //  ArrayList<Node> nodes; -- inherited from List
034    
035        /**
036         * Every <b>DatalogSegment</b> has one of the following four names:
037         * <i>Schemes</i>, <i>Facts</i>, <i>Rules</i>, or <i>Queries</i>.  
038         * The name is set when the specialization is cretaed.  This name is
039         * used for printing purposes.
040         * @see
041         *   <a href=
042         *      "../../srchtml/project3/DatalogSegment.html#name"
043         *      >Source Code
044         *   </a>
045         */
046        protected String name;
047    
048    //Constructor
049        /**
050         * Creates a DatalogSegment.
051         *
052         * <p>
053         * <table>
054         *   <tr>
055         *     <td valign=top>
056         *       <b>Precondition</b>:
057         *     </td>
058         *     <td valign=top>
059         *       lex &ne; null <br/>
060         *       lex.front().getTokenType() = expedtedTokenType <br/>
061         *       lex.popFront().front().getTokenType() = TokenType.COMMA <br/>
062         *       everything thereafter must be syntatically correct.
063         *     </td>
064         *     <td valign=top>
065         *       AND <br/>
066         *       AND <br/>
067         *       &nbsp;
068         *     </td>
069         *   </tr>
070         *   <tr>
071         *     <td valign=top>
072         *       <b>Postcondition</b>:
073         *     </td>
074         *     <td valign=top>
075         *       name = lex'.front().getValue() <br/>
076         *       createList(lex'[1..|lex'|-1].Postcondition
077         *     </td>
078         *     <td valign=top>
079         *       AND <br/>
080         *       &nbsp;
081         *     </td>
082         *   </tr>
083         * </table>
084         *
085         * @param lex the lexical analyzer we will read tokens from.
086         * @param expectedTokenType the type of the token we expect to see at
087         *                            lex[0]
088         *
089         * @throws ParserException
090         * @see
091         *   <a href=
092         *      "../../srchtml/project3/DatalogSegment.html#StdConstructor"
093         *      >Source Code
094         *   </a>
095         */
096        public DatalogSegment(Lex lex, TokenType expectedTokenType)
097            throws ParserException 
098        {
099            Token token = checkFor(lex, expectedTokenType);
100            name = token.getValue();
101            checkFor(lex, TokenType.COLON);
102            createList(lex);
103        }
104    
105    //Queries
106        /**
107         * Creates a lexical version of a DatalogSegment.
108         *
109         * <p>
110         * <table>
111         *   <tr>
112         *     <td valign=top>
113         *       <b>Precondition</b>:
114         *     </td>
115         *     <td valign=top>
116         *       none
117         *     </td>
118         *     <td valign=top>
119         *     </td>
120         *   </tr>
121         *   <tr>
122         *     <td valign=top>
123         *       <b>Postcondition</b>:
124         *     </td>
125         *     <td valign=top>
126         *       result = name + ":" + super.toString()
127         *     </td>
128         *     <td valign=top>
129         *     </td>
130         *   </tr>
131         * </table>
132         *
133         * @return The String representation of this DatalogSegment.
134         * @see
135         *   <a href=
136         *      "../../srchtml/project3/DatalogSegment.html#toString"
137         *      >Source Code
138         *   </a>
139         */
140        public String toString()
141        {
142            StringBuffer result = new StringBuffer();
143            result.append(name);
144            result.append(":"); 
145            result.append(System.getProperty("line.separator"));
146            result.append(super.toString());
147            return result.toString();
148        }
149    }
