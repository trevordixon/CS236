087        /**
088         * Creates a map of the variables in nodes to the constants in predicate.
089         *
090         * <p>
091         * <table>
092         *   <tr>
093         *     <td valign=top>
094         *       <b>Precondition</b>:
095         *     </td>
096         *     <td valign=top>
097         *       predicate &ne; null 
098         *     </td>
099         *     <td valign=top>
100         *       AND
101         *     </td>
102         *   </tr>
103         *   <tr>
104         *     <td valign=top>
105         *       &nbsp;
106         *     </td>
107         *     <td valign=top>
108         *       |predicate| = |nodes|
109         *     </td>
110         *     <td valign=top>
111         *       AND
112         *     </td>
113         *   </tr>
114         *   <tr>
115         *     <td valign=top>
116         *       &nbsp;
117         *     </td>
118         *     <td valign=top>
119         *       &forall;<sub>i</sub> (0 &lt;= i AND i &lt; |predicate| &rArr;
120         *           (predicate[i] <i>instanceof</i> <b>Constant</b>))
121         *     </td>
122         *     <td valign=top>
123         *     </td>
124         *   </tr>
125         *   <tr>
126         *     <td valign=top>
127         *       <b>Postcondition</b>:
128         *     </td>
129         *     <td valign=top>
130         *       |result| = |nodes|
131         *     </td>
132         *     <td valign=top>
133         *       AND
134         *     </td>
135         *   </tr>
136         *   <tr>
137         *     <td valign=top>
138         *       &nbsp;
139         *     </td>
140         *     <td valign=top>
141         *       &forall;<sub>n</sub> (
142         *           &forall;<sub>i</sub> (
143         *               (result.getKeys().contains(n) AND
144         *                    nodes[i] = n &rArr; predicate[i] = result.get(n))))
145         *     </td>
146         *     <td valign=top>
147         *     </td>
148         *   </tr>
149         * </table>
150         *
151         * @param predicate the predicate we will unify with this <b>Head</b>.
152         *
153         * @return a mapping from each of the variables in the head to the
154         *         constants in predicate.
155         * @see
156         *   <a href=
157         *      "../../srchtml/project3/Head.html#unify"
158         *      >Source Code
159         *   </a>
160         */
161        public HashMap<Identifier, Constant> unify(Predicate predicate)
162        {
163            assert predicate != null;
164            assert predicate.size() == this.size();
165            assert allConstants(predicate);
166    
167            HashMap<Identifier, Constant> result =
168                new HashMap<Identifier, Constant>();
169    
170            for(int i = 0; i < nodes.size(); i++){
171                result.put((Identifier)nodes.get(i),
172                           (Constant)predicate.getNodes().get(i));            
173            };
174    
175            return result;
176        }
177    
178        /**
179         * Checks to see if this <b>Head</b> matches the given <b>PredicateList</b>.
180         *
181         * <p>
182         * <table>
183         *   <tr>
184         *     <td valign=top>
185         *       <b>Precondition</b>:
186         *     </td>
187         *     <td valign=top>
188         *       predicateList &ne; null
189         *     </td>
190         *     <td valign=top>
191         *     </td>
192         *   </tr>
193         *   <tr>
194         *     <td valign=top>
195         *       <b>Postcondition</b>:
196         *     </td>
197         *     <td valign=top>
198         *       result = (name = predicateList.name AND |this| = |predicateList|)
199         *     </td>
200         *     <td valign=top>
201         *     </td>
202         *   </tr>
203         * </table>
204         *
205         * @param predicate the predicate we are going to try and match.
206         *
207         * @return whether the predicates match or not.
208         * @see
209         *   <a href=
210         *      "../../srchtml/project3/Head.html#matches"
211         *      >Source Code
212         *   </a>
213         */
214        public boolean matches(Predicate predicate)
215        {
216            boolean result = name.equals(predicate.getName()) &&
217                             nodes.size() == predicate.getNodes().size();
218            return result;
219        }

224        /**
225         * Check to see if all of the arguments in the predicate are constants.
226         *
227         * <p>
228         * <table>
229         *   <tr>
230         *     <td valign=top>
231         *       <b>Precondition</b>:
232         *     </td>
233         *     <td valign=top>
234         *       predicate &ne; null
235         *     </td>
236         *     <td valign=top>
237         *     </td>
238         *   </tr>
239         *   <tr>
240         *     <td valign=top>
241         *       <b>Postcondition</b>:
242         *     </td>
243         *     <td valign=top>
244         *       result = &forall; <sub>i</sub>
245         *         (0 &lt;= i AND i &lt; |predicate| &rArr;
246         *                    predicate[i] <i>instanceof</i> <b>Constant</b>)
247         *     </td>
248         *     <td valign=top>
249         *     </td>
250         *   </tr>
251         * </table>
252         *
253         * @param predicate the predicate we are going to examine to see if it
254         *                    contains all constants.
255         *
256         * @return whether the provided predicate contains all constants.
257         * @see
258         *   <a href=
259         *      "../../srchtml/project3/Head.html#allConstants"
260         *      >Source Code
261         *   </a>
262         */
263        private boolean allConstants(Predicate predicate)
264        {
265            assert predicate != null;
266            boolean result = true;
267            Iterator iter = predicate.iterator();
268            while(iter.hasNext() && result){
269                result = iter.next() instanceof Constant;
270            };
271            return result;
272        }
