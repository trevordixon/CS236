011    /**
012     * A <b>Query</b> in the datalog program.
013     *
014     * <pre>
015     * Syntax: <b>Query</b> &rarr; <b>Predicate</b> "?"
016     *
017     * Domain Definition:
018     *    predicates : <b>Predicate</b>; 
019     *
020     *    setOfSolutions : <b>SetOfSolutions</b>
021     *    -- A list of arrays of constants that represent the valid substitutions
022     *    -- for the variables in the variable list. There is a one-to-one
023     *    -- correspondence between the constants in an array and the variables in
024     *    -- the query.
025     *
026     *    variableInformation: <b>QueryVariableInformation</b>
027     *    -- The set of variables in the Query, and, for each variable the 
028     *    -- corresponding locations in the Query and current value of that
029     *    -- variable.  
030     *
031     *    variableIterator : Iterator<Identifier>
032     *    -- An iterator over the variableInformation.  Inherited from 
033     *    -- <b>PredicateList</b>
034     *
035     *    Instance Invariant:
036     *        |predicates| = 1
037     * </pre>
038     *
039     * Last Modified: 21 Feb 2006
040     *
041     * @author Scott N. Woodfield
042     * @version 1.0
043     * @see
044     *   <a href=
045     *      "../../srchtml/project3/Query.html"
046     *      >Source Code
047     *   </a>
048     */
049    public class Query
050        extends PredicateList
051    {
052    //Domain Implementation
053        // VariableInformation variableInformation
054        //   -- inherited from PredicateList. But, we assign an instance of
055        //   -- QueryVariableInformation to it.  A QueryVariableInformation inherits
056        //   -- everything from VariableInformation plus it includes the current
057        //   -- value for a variable.
058        //    
059        // Iterator<Identifier> variableIterator -- inherited from PredicateList
060    
061        /**
062         * A list of <b>Solution</b>s where a <b>Solution</b> is an array of
063         * <i>Constant</i>s such that the constants in the <b>Solution</b> have a
064         * one-to-one correspondence with the variables in the query.
065         * @see
066         *   <a href=
067         *      "../../srchtml/project3/Query.html#setOfSolutions"
068         *      >Source Code
069         *   </a>
070         */
071        protected SetOfSolutions setOfSolutions = null;
072    
073    //Constructors
074        /**
075         * The lex-based constructor used to create a Query from the lexical
076         * analyzer.
077         * 
078         * <p>
079         * <table>
080         *   <tr>
081         *     <td valign=top>
082         *       <b>Precondition</b>:
083         *     </td>
084         *     <td valign=top>
085         *       lex &ne; null
086         *     </td>
087         *     <td valign=top>
088         *       AND
089         *     </td>
090         *   </tr>
091         *   <tr>
092         *     <td valign=top>
093         *       &nbsp;
094         *     </td>
095         *     <td valign=top>
096         *       <b>PredicateList</b>.Precondition 
097         *     </td>
098         *     <td valign=top>
099         *     </td>
100         *   </tr>
101         *   <tr>
102         *     <td valign=top>
103         *       &nbsp;
104         *     </td>
105         *     <td valign=top>
106         *       |<i>nodes</i>| = 1
107         *     </td>
108         *     <td valign=top>
109         *     </td>
110         *   </tr>
111         *   <tr>
112         *     <td valign=top>
113         *       <b>Postcondition</b>:
114         *     </td>
115         *     <td valign=top>
116         *         <b>PredicateList</b>.Postcondition 
117         *     </td>
118         *     <td valign=top>
119         *       AND
120         *     </td>
121         *   </tr>
122         *   <tr>
123         *     <td valign=top>
124         *       &nbsp;
125         *     </td>
126         *     <td valign=top>
127         *         setOfSolutions =
128         *            new SetOfSolutions(variableInformation.getVariables())
129         *     </td>
130         *     <td valign=top>
131         *     </td>
132         *   </tr>
133         * </table>
134         *
135         * @throws ParserException
136         *
137         * @param lex the lexical analyzer we will read from to create the
138         *              <b>Query</b>
139         * @see
140         *   <a href=
141         *      "../../srchtml/project3/Query.html#LexConstructor"
142         *      >Source Code
143         *   </a>
144         */
145        public Query(Lex lex)
146            throws ParserException
147        {
148            super(lex);
149    
150            checkFor(lex, TokenType.Q_MARK);
151    
152            if(nodes.size() != 1)
153                throw new ParserException(
154                    "Error while parsing Query on or before line " +
155                    lex.front().getLineNumber() + ".\n" +
156                    "A Query can have only one simple predicate");
157    
158            setOfSolutions =
159                new SetOfSolutions(
160                    ((QueryVariableInformation)variableInformation).getVariables());
161        }
162    
163    //Queries
164        /**
165         * Evaluates the query for all combinations of constants from the domain,
166         * for all variables in the query.
167         *
168         * <p>
169         * <table>
170         *   <tr>
171         *     <td valign=top>
172         *       <b>Precondition</b>:
173         *     </td>
174         *     <td valign=top>
175         *       PredicateList.evaluate().Precondition 
176         *     </td>
177         *     <td valign=top>
178         *     </td>
179         *   </tr>
180         *   <tr>
181         *     <td valign=top>
182         *       <b>Postcondition</b>:
183         *     </td>
184         *     <td valign=top>
185         *       PredicateList.evaluate().PostCondition 
186         *     </td>
187         *     <td valign=top>
188         *       AND
189         *     </td>
190         *   </tr>
191         *   <tr>
192         *     <td valign=top>
193         *       &nbsp;
194         *     </td>
195         *     <td valign=top>
196         *       strBuffer = strBuffer + toString()
197         *     </td>
198         *     <td valign=top>
199         *     </td>
200         *   </tr>
201         * </table>
202         *
203         * @throws ParserException
204         *
205         * @param strBuffer the buffer in which we store the result.
206         *
207         * @return true iff we found a substitution for the variables in the
208         *         <b>Query</b>
209         * @see
210         *   <a href=
211         *      "../../srchtml/project3/Query.html#evaluate"
212         *      >Source Code
213         *   </a>
214         */
215        public boolean evaluate(StringBuffer strBuffer)
216            throws ParserException
217        {
218           //We copy <i>this</i> query as a PredicateList so that when apply
219           //"toString()" to it we will get the "toString()" method of
220           //PredicateList rather than the "toString()" method of Query.
221            copy = new PredicateList(this);
222    
223            boolean result = super.evaluate();
224            strBuffer.append(toString());
225            return result;
226        }
227    
228        /**
229         * Returns the lexical form of the query plus the results of the query.
230         *
231         * <p>
232         * <table>
233         *   <tr>
234         *     <td valign=top>
235         *       <b>Precondition</b>:
236         *     </td>
237         *     <td valign=top>
238         *       none
239         *     </td>
240         *     <td valign=top>
241         *     </td>
242         *   </tr>
243         *   <tr>
244         *     <td valign=top>
245         *       <b>Postcondition</b>:
246         *     </td>
247         *     <td valign=top>
248         *     copy = null &rArr;
249         *         reuslt = super.toString() + setOfSolutions.toString() + "\n"
250         *     </td>
251         *     <td valign=top>
252         *       AND
253         *     </td>
254         *   </tr>
255         *   <tr>
256         *     <td valign=top>
257         *       &nbsp;
258         *     </td>
259         *     <td valign=top>
260         *       copy &ne; null &rArr;
261         *         reuslt = copy.toString() + setOfSolutions.toString() + "\n"
262         *     </td>
263         *     <td valign=top>
264         *     </td>
265         *   </tr>
266         * </table>
267         *
268         * @return the lexical representation of the <b>Query</b>
269         * @see
270         *   <a href=
271         *      "../../srchtml/project3/Query.html#toString"
272         *      >Source Code
273         *   </a>
274         */
275        public String toString()
276        {
277            StringBuffer result = new StringBuffer();
278            if(copy == null){
279                result.append(super.toString());
280            } else {
281                result.append(copy.toString());
282            };
283            result.append("?");
284            result.append(setOfSolutions);
285            //result.append("\n");
286            return result.toString();
287        }

290        /**
291         * Saves all of the valid solutions.
292         *
293         * <p>
294         * <table>
295         *   <tr>
296         *     <td valign=top>
297         *       <b>Precondition</b>:
298         *     </td>
299         *     <td valign=top>
300         *       none
301         *     </td>
302         *     <td valign=top>
303         *     </td>
304         *   </tr>
305         *   <tr>
306         *     <td valign=top>
307         *       <b>Postcondition</b>:
308         *     </td>
309         *     <td valign=top>
310         *       setOfSolutions =
311         *          setOfSolutions' + variableInformation.getSolution()
312         *     </td>
313         *     <td valign=top>
314         *     </td>
315         *   </tr>
316         * </table>
317         * @see
318         *   <a href=
319         *      "../../srchtml/project3/Query.html#saveResult"
320         *      >Source Code
321         *   </a>
322         */
323        protected void saveResult()
324        {
325            setOfSolutions.addSolution(
326                ((QueryVariableInformation)variableInformation).getSolution());
327        }

329        /**
330         * Everyplace in the predicate list where <i>variable</i> appears, it is
331         * replaced by the given value and we remember what that value was so
332         * we can print it later.
333         *
334         * <p>
335         * <table>
336         *   <tr>
337         *     <td valign=top>
338         *       <b>Precondition</b>:
339         *     </td>
340         *     <td valign=top>
341         *       variable &ne; null
342         *     </td>
343         *     <td valign=top>
344         *       AND
345         *     </td>
346         *   </tr>
347         *   <tr>
348         *     <td valign=top>
349         *       &nbsp;
350         *     </td>
351         *     <td valign=top>
352         *       value &ne; null
353         *     </td>
354         *     <td valign=top>
355         *     </td>
356         *   </tr>
357         *   <tr>
358         *     <td valign=top>
359         *       <b>Postcondition</b>:
360         *     </td>
361         *     <td valign=top>
362         *        Everyplace where variable appeared in the predicateList
363         *        it has been replaced by the given value.
364         *     </td>
365         *     <td valign=top>
366         *     </td>
367         *   </tr>
368         * </table>
369         *
370         * @throws ParserException
371         *
372         * @param variable all variables with this name are changed to the given
373         *                   value.
374         * @param value the value we are going to change the variable(s) to.
375         * @see
376         *   <a href=
377         *      "../../srchtml/project3/Query.html#setVariableToValue"
378         *      >Source Code
379         *   </a>
380         */
381        public void setVariableToValue(Identifier variable, Constant value)
382            throws ParserException
383        {
384            super.setVariableToValue(variable, value);
385            //assert variable != null || value != null; -- checked in superclass
386            ((QueryVariableInformation)variableInformation).
387                setValueFor(variable, value);
388        }

460        /**
461         * Should we continue finding all combinations of constants?  If we are
462         * recursing over a predicate list, then we can stop the first time
463         * we find a true one,  if this is a query, then we never stop.  Since this
464         * is a query we never stop.
465         *
466         * <p>
467         * <table>
468         *   <tr>
469         *     <td valign=top>
470         *       <b>Precondition</b>:
471         *     </td>
472         *     <td valign=top>
473         *       None. Ignore validSolutionFound. 
474         *     </td>
475         *     <td valign=top>
476         *     </td>
477         *   </tr>
478         *   <tr>
479         *     <td valign=top>
480         *       <b>Postcondition</b>:
481         *     </td>
482         *     <td valign=top>
483         *       result = true
484         *     </td>
485         *     <td valign=top>
486         *     </td>
487         *   </tr>
488         * </table>
489         *
490         * @param validSolutionFound indicates whether or not a valid
491         *                             solution was found.  In this case we
492         *                             don't care but in <b>PredicateList</b> we do.
493         *
494         * @return true always -- for <b>Query</b>s we try all possible
495         *                        combinations.
496         * @see
497         *   <a href=
498         *      "../../srchtml/project3/Query.html#keepOnGoing"
499         *      >Source Code
500         *   </a>
501         */
502        protected boolean keepOnGoing(boolean validSolutionFound)
503        {
504            return true;
505        }
